declare
v_session_user varchar2(200);
v_session_user_id varchar2(20);
os_user varchar2(20);
v_ip_address varchar2(20);
begin
v_session_user := sys_context('USERENV','SESSION_USER');
v_session_user_id := sys_context('USERENV','SESSION_USERID');
os_user := sys_context('USERENV','OS_USER');
v_ip_address := sys_context('USERENV','IP_ADDRESS');
dbms_output.put_line(v_session_user);
dbms_output.put_line(v_session_user_id);
dbms_output.put_line(os_user);
dbms_output.put_line(v_ip_address);
end;
